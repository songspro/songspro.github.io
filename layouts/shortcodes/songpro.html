{{- range $lines -}} {{- $line := . -}} {{- if or (hasPrefix $line "@") (hasPrefix $line "!") -}}
{{ $line }}
{{- else if hasPrefix $line "# " -}}
{{ $line }}
{{- else -}}
{{ $line | replaceRE "\\[([^]]+)\\]" "[$1]" | safeHTML }} 
{{- end -}} {{- end -}}
